### Solution :

- Time Complexity : 


```java
public int LongestBitonicSequence(int[] nums)
{
    int N = nums.length;
    int [] LIS = new int[N];
    int [] LDS = new int[N];

    for(int i=0; i< N; i++){
        LIS[i] = 1;
        for(int j=0; j< i; j++){
            if(nums[i] > nums[j] ){
                if(LIS[j]+1 > LIS[i]) LIS[i] = LIS[j] + 1;
            }
        }
    }

    for(int i= N-1; i>=0; i--){
        LDS[i] = 1;
        for(int j= N-1; j> i; j--){
            if(nums[i] > nums[j]){
                if(LDS[j]+1 > LDS[i]) LDS[i] = LDS[j] + 1;
            }
        }
    }

    int Max = 0;
    for(int i=0; i< N; i++){
        Max = Math.max(Max, LIS[i]+ LDS[i] -1);
    }
    return Max;
}

```
